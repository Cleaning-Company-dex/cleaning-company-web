"""
Initialize sheets in existing spreadsheet
"""

import gspread
from google.oauth2.service_account import Credentials
from datetime import datetime
import os
import sys

# Add parent directory to path to find credentials.json
sys.path.append(os.path.dirname(os.path.dirname(os.path.abspath(__file__))))

def initialize_sheets():
    print("ðŸ”§ Setting up sheets in existing spreadsheet...")
    print("=" * 50)
    
    # Setup credentials - look in parent directory
    scope = ['https://spreadsheets.google.com/feeds',
            'https://www.googleapis.com/auth/drive']
    
    creds = Credentials.from_service_account_file('../credentials.json', scopes=scope)
    client = gspread.authorize(creds)
    print("âœ… Connected to Google Sheets")
    
    # Open existing spreadsheet
    try:
        spreadsheet = client.open('Cleaning_Business_Database')
        print("âœ… Found spreadsheet: Cleaning_Business_Database")
    except Exception as e:
        print(f"âŒ Error: {e}")
        print("Make sure you:")
        print("1. Created a spreadsheet named 'Cleaning_Business_Database'")
        print("2. Shared it with your service account email")
        return False
    
    # Define sheet structures
    sheets_structure = {
        'Customers': ['ID', 'Name', 'Email', 'Phone', 'Address', 'Business_Type',
                     'Square_Feet', 'Service_Frequency', 'Added_Date', 'Status',
                     'Special_Instructions', 'Preferred_Day', 'Preferred_Time'],
        'Jobs': ['ID', 'Customer_Name', 'Date', 'Time', 'Employee', 'Address',
                'Service_Type', 'Price', 'Status', 'Completed', 'Notes',

    }

    existing_sheets = [sheet.title for sheet in spreadsheet.worksheets()]

            print(f"âœ… Created sheet: {sheet_name}")
        else:
            worksheet = spreadsheet.worksheet(sheet_name)
            if not worksheet.row_values(1):
                worksheet.append_row(headers)
                print(f"âœ… Added headers to: {sheet_name}")
            else:
                print(f"âœ“ Sheet exists: {sheet_name}")
    
    # Delete default Sheet1 if exists
    try:
        default_sheet = spreadsheet.worksheet('Sheet1')

        pass
